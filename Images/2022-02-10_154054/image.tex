    \begin{tikzpicture}[scale=0.75, color=textColor]
        %Macros
            \pgfmathsetmacro{\xScale}{4}
            \pgfmathsetmacro{\yScale}{6}

            \pgfmathsetmacro{\a}{0}
            \pgfmathsetmacro{\b}{1*\yScale/6}
            \pgfmathsetmacro{\c}{\xScale}
            \pgfmathsetmacro{\d}{5*\yScale/6}
            
            \pgfmathsetmacro{\art}{\d+1}
            \pgfmathsetmacro{\arb}{\b-1}

            \pgfmathsetmacro{\f}{\xScale/4}
            \pgfmathsetmacro{\hs}{-\f}
            \pgfmathsetmacro{\he}{\xScale+\f}

            \pgfmathsetmacro{\ra}{0.4}

        %Grid
            \draw[thin, dashed] (\a,0) -- (\a,\yScale);
            \draw[thin, dashed] (\c,0) -- (\c,\yScale);
            \draw[thin, dashed] (\hs,\b) -- (\he,\b);
            \draw[thin, dashed] (\hs,\d) -- (\he,\d);
            \draw[thin, dashed, color=red] (\hs,3.5) -- (\he,3.5);
            \draw[thin, dashed, color=green] (\hs,1.5) -- (\he,1.5);

        %Labels
            \draw (\a,-0.25) circle (0) node {\footnotesize{$a$}};
            \draw (\c,-0.25) circle (0) node {\footnotesize{$c$}};
            \draw (\hs-0.25,\b) circle (0) node {\footnotesize{$b$}};
            \draw (\hs-0.25,\d) circle (0) node {\footnotesize{$d$}};
            \draw (\hs-0.25,3.5) circle (0) node {\color{red}\footnotesize{$z$}};
            \draw (\hs-0.25,1.5) circle (0) node {\color{green}\footnotesize{$z$}};

        %Side arrows
            \draw[thick, (->] (\a,\b) -- (\a,\art);
            \draw[thick, (->] (\c,\d) -- (\c,\arb);

        % Main intervals
            \fill[color=red] (\a,3) circle (0.05) node[right] {\color{red}\footnotesize{$\l\langle a,y\r\rangle$}};
            \draw[(-), thick, color=red] (\a,\b) -- (\a,3.5);
            \fill[color=green] (\c,2.5) circle (0.05) node[left] {\color{green}\footnotesize{$\l\langle c,y\r\rangle$}};
            \draw[(-), thick, color=green] (\c,\d) -- (\c,1.5);
            \fill[color=blue] (\c/3,4.25) circle (0.05) node[right] {\color{blue}\footnotesize{$\l\langle x,y\r\rangle$}};
            \draw[(-), thick, color=blue] (\c/3,4.25+\ra) -- (\c/3,4.25-\ra);
    \end{tikzpicture}
